postMessage — Missing Origin Check

This page hosts a widget that receives postMessage events without verifying event.origin. Any window — including one on a different origin — can send it a getProfile message and receive the flag in the reply. A render message triggers DOM-XSS by setting innerHTML directly.

📜 Vulnerable Listener Code 
// This code runs on the victim page (x3.html)

window.__FLAG__ = 'HUNT{postmessage_needs_an_origin_check}';

window.addEventListener('message', (event) => {
  // BUG: no event.origin check — any window can send messages
  // SAFE version: if (event.origin !== 'http://localhost:3000') return;

  let msg;
  try { msg = JSON.parse(event.data); }
  catch { return; }

  if (msg.type === 'getProfile') {
    // BUG: sends flag to any origin using targetOrigin '*'
    event.source.postMessage(
      JSON.stringify({ flag: window.__FLAG__, user: 'victim' }),
      '*'   ← should be the expected origin, not '*'
    );
  }

  if (msg.type === 'render') {
    // BUG: DOM-XSS — unsanitised HTML from any message sender
    document.getElementById('render-target').innerHTML = msg.html;
  }
});
